import java.io.*;
import java.net.*;

public class MyServerCalc {
    public static void main(String args[]) throws Exception {
        ServerSocket ss = new ServerSocket(3333);
        Socket s = ss.accept();

        DataInputStream din = new DataInputStream(s.getInputStream());
        DataOutputStream dout = new DataOutputStream(s.getOutputStream());
        BufferedReader br = new BufferedReader(new InputStreamReader(System.in));

        String str = "", str2 = "Infinity";
        int a, b, res, flag = 1;

        while (!str.equals("Stop")) {
            res = 0;
            flag = 0;
            str = din.readUTF();

            if (str.equals("Stop")) {
                System.out.println("Operation stopped");
                break;
            }

            if (str.equals("stop")) {
                System.out.println("Operation stopped");
                break;
            }

            a = Integer.parseInt(din.readUTF());
            b = Integer.parseInt(din.readUTF());

            if (str.equals("Addition")) {
                res = a + b;
            } else if (str.equals("Subtraction")) {
                res = a - b;
            } else if (str.equals("Multiplication")) {
                res = a * b;
            } else if (str.equals("Division")) {
                try {
                    if (b != 0) {
                        res = a / b;
                    } else {
                        System.out.println("Division by zero is not possible");
                        dout.writeUTF(str2);
                        dout.flush();
                        flag = 1;
                    }
                } catch (Exception e) {
                    e.printStackTrace();
                }
            }

            if (flag == 0) {
                dout.writeUTF(Integer.toString(res));
                dout.flush();
            }
        }

        din.close();
        dout.close();
        br.close();
        s.close();
        ss.close();
    }
}
